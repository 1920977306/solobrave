### 任务1：侧栏权限过滤改造 🔴 最高优先级

**当前代码** index.html 约 L4432：
```javascript
function getVisibleEmps(){
  var uid = currentUser && currentUser.id;
  var isAdmin = currentUser && currentUser.role === 'admin';
  if(isAdmin || !uid) return emps;
  return emps.filter(function(e){ return e.createdBy === uid; });
}
```

**改造目标**：
- admin → 看到所有员工（不变）
- leader → 看到本组+子组所有成员的员工
- employee → 只看到自己创建的员工

**实现方案**：
1. 改造 `getVisibleEmps()` 函数，新增leader分支
2. leader登录后调 `GET /api/teams` 获取可见小组列表，缓存到 `window._visibleTeams`
3. 根据小组的 `agentIds` 过滤员工列表

```javascript
function getVisibleEmps(){
  if(!emps || emps.length === 0) return [];
  var uid = currentUser && currentUser.id;
  var role = currentUser && currentUser.role;
  if(role === 'admin' || !uid) return emps;
  if(role === 'leader'){
    var visibleIds = _getLeaderVisibleAgentIds();
    return emps.filter(function(e){ return visibleIds.indexOf(e.id) >= 0 || e.createdBy === uid; });
  }
  return emps.filter(function(e){ return e.createdBy === uid; });
}

function _getLeaderVisibleAgentIds(){
  var teams = window._visibleTeams || [];
  var ids = [];
  teams.forEach(function(t){
    if(t.agentIds){
      t.agentIds.forEach(function(aid){
        if(ids.indexOf(aid) < 0) ids.push(aid);
      });
    }
  });
  return ids;
}
```

**验收**：admin看全部、leader看本组+子组、employee只看自己的

function requireRole(requiredRole) {
  if (!currentUser) return false;
  if (requiredRole === 'admin') return currentUser.role === 'admin';
  if (requiredRole === 'leader') return currentUser.role === 'admin' || currentUser.role === 'leader';
